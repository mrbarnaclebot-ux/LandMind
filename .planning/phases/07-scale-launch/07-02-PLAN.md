import { Connection, ComputeBudgetProgram, TransactionInstruction } from '@solana/web3.js';

// Base priority fee in microLamports
const BASE_PRIORITY_FEE = 1000;
// Max priority fee cap (prevent overpayment)
const MAX_PRIORITY_FEE = 1_000_000; // 1M microLamports = 0.001 SOL per CU

export async function estimatePriorityFee(
  connection: Connection,
  retryAttempt: number
): Promise<number> {
  // Exponential escalation: 1000 * 2^attempt
  // Attempt 0: 1000, 1: 2000, 2: 4000, 3: 8000, 4: 16000
  const fee = BASE_PRIORITY_FEE * Math.pow(2, retryAttempt);
  return Math.min(fee, MAX_PRIORITY_FEE);
}

export function createPriorityFeeInstruction(microLamports: number): TransactionInstruction {
  return ComputeBudgetProgram.setComputeUnitPrice({ microLamports });
}

export function createComputeUnitLimitInstruction(units: number): TransactionInstruction {
  return ComputeBudgetProgram.setComputeUnitLimit({ units });
}

import {
  Connection,
  Transaction,
  VersionedTransaction,
  Signer,
  SendOptions
} from '@solana/web3.js';
import { estimatePriorityFee, createPriorityFeeInstruction } from './priorityFees';

export interface RetryConfig {
  maxRetries?: number;           // Default: 5
  initialBackoffMs?: number;     // Default: 500
  onRetry?: (attempt: number, priorityFee: number) => void;
  onStatusChange?: (status: TransactionStatus) => void;
}

export type TransactionStatus =
  | { status: 'sending' }
  | { status: 'confirming'; signature: string }
  | { status: 'retrying'; attempt: number; priorityFee: number }
  | { status: 'confirmed'; signature: string }
  | { status: 'expired' }
  | { status: 'failed'; error: string };

export async function sendTransactionWithRetry(
  connection: Connection,
  transaction: Transaction,
  signers: Signer[],
  config: RetryConfig = {}
): Promise<string> {
  const {
    maxRetries = 5,
    initialBackoffMs = 500,
    onRetry,
    onStatusChange
  } = config;

  // Get blockhash with lastValidBlockHeight for expiration tracking
  const { blockhash, lastValidBlockHeight } =
    await connection.getLatestBlockhash('confirmed');

  transaction.recentBlockhash = blockhash;
  transaction.feePayer = signers[0].publicKey;

  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      // Clone transaction for modification on retry
      const txToSend = Transaction.from(transaction.serialize({ requireAllSignatures: false }));

      // Add priority fee on retries
      if (attempt > 0) {
        const priorityFee = await estimatePriorityFee(connection, attempt);
        const priorityIx = createPriorityFeeInstruction(priorityFee);
        txToSend.instructions = [priorityIx, ...txToSend.instructions];
        txToSend.recentBlockhash = blockhash; // Re-set after modification

        onRetry?.(attempt, priorityFee);
        onStatusChange?.({ status: 'retrying', attempt, priorityFee });
      } else {
        onStatusChange?.({ status: 'sending' });
      }

      // Sign transaction
      txToSend.sign(...signers);

      // Send with maxRetries=0 for manual control
      const signature = await connection.sendRawTransaction(
        txToSend.serialize(),
        { skipPreflight: false, maxRetries: 0 }
      );

      onStatusChange?.({ status: 'confirming', signature });

      // Poll for confirmation until blockhash expires
      const confirmation = await confirmTransactionUntilExpiry(
        connection,
        signature,
        lastValidBlockHeight
      );

      if (confirmation.err) {
        throw new Error(`Transaction failed: ${JSON.stringify(confirmation.err)}`);
      }

      onStatusChange?.({ status: 'confirmed', signature });
      return signature;

    } catch (error) {
      // Check if blockhash expired
      const currentHeight = await connection.getBlockHeight('confirmed');
      if (currentHeight > lastValidBlockHeight) {
        onStatusChange?.({ status: 'expired' });
        throw new Error('Transaction expired - blockhash too old');
      }

      // Last attempt - throw error
      if (attempt === maxRetries - 1) {
        const errorMsg = error instanceof Error ? error.message : 'Unknown error';
        onStatusChange?.({ status: 'failed', error: errorMsg });
        throw error;
      }

      // Exponential backoff before retry
      await new Promise(resolve =>
        setTimeout(resolve, initialBackoffMs * Math.pow(2, attempt))
      );
    }
  }

  throw new Error('Max retries exceeded');
}

async function confirmTransactionUntilExpiry(
  connection: Connection,
  signature: string,
  lastValidBlockHeight: number,
  pollIntervalMs = 500
): Promise<{ err: any }> {
  while (true) {
    // Check block height first
    const currentHeight = await connection.getBlockHeight('confirmed');
    if (currentHeight > lastValidBlockHeight) {
      throw new Error('Blockhash expired before confirmation');
    }

    // Check transaction status
    const response = await connection.getSignatureStatus(signature);

    if (response?.value?.confirmationStatus === 'confirmed' ||
        response?.value?.confirmationStatus === 'finalized') {
      return { err: response.value.err };
    }

    await new Promise(resolve => setTimeout(resolve, pollIntervalMs));
  }
}

```typescript
import { create } from 'zustand';

export interface TransactionToast {
  id: string;
  type: 'info' | 'success' | 'warning' | 'error';
  title: string;
  message: string;
  signature?: string;
  autoHide?: number; // ms, undefined = manual dismiss
}

interface TransactionStore {
  toasts: TransactionToast[];
  addToast: (toast: Omit<TransactionToast, 'id'>) => string;
  removeToast: (id: string) => void;
  clearToasts: () => void;
}

export const useTransactionStore = create<TransactionStore>((set, get) => ({
  toasts: [],
  addToast: (toast) => {
    const id = crypto.randomUUID();
    set(state => ({ toasts: [...state.toasts, { ...toast, id }] }));

    if (toast.autoHide) {
      setTimeout(() => get().removeToast(id), toast.autoHide);
    }
    return id;
  },
  removeToast: (id) => {
    set(state => ({ toasts: state.toasts.filter(t => t.id !== id) }));
  },
  clearToasts: () => set({ toasts: [] }),
}));
```

export function useTransactionToast() {
  const { addToast, removeToast } = useTransactionStore();

  return {
    showSending: () => addToast({
      type: 'info',
      title: 'Sending',
      message: 'Sending transaction...',
    }),
    showConfirming: (signature: string) => addToast({
      type: 'info',
      title: 'Confirming',
      message: 'Waiting for confirmation...',
      signature,
    }),
    showRetrying: (attempt: number, fee: number) => addToast({
      type: 'warning',
      title: 'Retrying',
      message: `Attempt ${attempt + 1} with priority fee ${fee} microLamports`,

    }),
    showSuccess: (signature: string) => addToast({
      type: 'success',
      title: 'Confirmed',

    showError: (error: string) => addToast({

    }),
    showExpired: () => addToast({
      type: 'error',
      title: 'Expired',
      message: 'Transaction expired. Please try again.',
      autoHide: 5000,
    }),
  };
}
