use anchor_lang::prelude::*;

#[account]
pub struct FeeVaultState {
    /// Admin authority who can pause/unpause and update merkle root
    pub authority: Pubkey,
    /// Current merkle root for claim verification
    pub merkle_root: [u8; 32],
    /// Total amount distributed (for tracking)
    pub total_distributed: u64,
    /// Whether claims are paused
    pub paused: bool,
    /// PDA bump for signing
    pub bump: u8,
}

impl FeeVaultState {
    // Account discriminator (8) + pubkey (32) + merkle_root (32) + u64 (8) + bool (1) + u8 (1) = 82
    // Add padding for safety = 128
    pub const SIZE: usize = 8 + 32 + 32 + 8 + 1 + 1 + 46;
}

// Events
#[event]
pub struct AgentDeployedEvent {

    pub timestamp: i64,

pub fn update_merkle_root(ctx: Context<AdminAction>, new_root: [u8; 32]) -> Result<()> {
    let old_root = ctx.accounts.vault_state.merkle_root;
    ctx.accounts.vault_state.merkle_root = new_root;

    emit!(MerkleRootUpdatedEvent {
        authority: ctx.accounts.authority.key(),
        old_root,
        new_root,
        timestamp: Clock::get()?.unix_timestamp,
    });

    msg!("Merkle root updated by {}", ctx.accounts.authority.key());
    Ok(())
}
