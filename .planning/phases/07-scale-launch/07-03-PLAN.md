import { useState, useEffect } from 'react';

export function useMediaQuery(query: string): boolean {
  const [matches, setMatches] = useState(() =>
    typeof window !== 'undefined' ? window.matchMedia(query).matches : false
  );

  useEffect(() => {
    const mediaQuery = window.matchMedia(query);
    const handler = (e: MediaQueryListEvent) => setMatches(e.matches);

    mediaQuery.addEventListener('change', handler);
    return () => mediaQuery.removeEventListener('change', handler);
  }, [query]);

  return matches;
}

export function useMobile() {
  const isMobile = useMediaQuery('(max-width: 768px)');
  const isTablet = useMediaQuery('(max-width: 1024px)');
  const isTouchDevice = useMediaQuery('(pointer: coarse)');

  return {
    isMobile,
    isTablet,
    isTouchDevice,
    isDesktop: !isMobile && !isTablet,
  };
}

import Sheet from 'react-modal-sheet';
import { ReactNode } from 'react';
import '../styles/mobile.css';

interface BottomSheetProps {
  isOpen: boolean;
  onClose: () => void;
  children: ReactNode;
  title?: string;
  snapPoints?: number[]; // Default: [0.9, 0.5, 0.25]
}

export function BottomSheet({
  isOpen,
  onClose,
  children,
  title,
  snapPoints = [0.9, 0.5, 0.25],
}: BottomSheetProps) {
  return (
    <Sheet
      isOpen={isOpen}
      onClose={onClose}
      snapPoints={snapPoints}
      initialSnap={1} // Start at middle snap
    >
      <Sheet.Container className="bottom-sheet-container">
        <Sheet.Header className="bottom-sheet-header">
          {title && <div className="bottom-sheet-title">{title}</div>}
          <div className="bottom-sheet-handle" />
        </Sheet.Header>
        <Sheet.Content className="bottom-sheet-content">
          <Sheet.Scroller>{children}</Sheet.Scroller>
        </Sheet.Content>
      </Sheet.Container>
      <Sheet.Backdrop onTap={onClose} />
    </Sheet>
  );
}

import { ReactNode, useState } from 'react';
import { BottomSheet } from './BottomSheet';
import { ConnectButton } from '../wallet/ConnectButton';

interface MobileLayoutProps {

  onLocateAgent: (q: number, r: number) => void;
}

type ActivePanel = 'none' | 'agents' | 'earnings' | 'settings';

export function MobileLayout({ children, onLocateAgent }: MobileLayoutProps) {
  const [activePanel, setActivePanel] = useState<ActivePanel>('none');

  return (
    <div className="mobile-layout">
      {/* Compact header */}
      <header className="mobile-header">
        <div className="mobile-header-logo">LANDMIND</div>

      {/* Main content */}
      <div style={{ paddingTop: '56px', paddingBottom: '60px', height: '100%' }}>
        {children}
      </div>

      {/* Bottom navigation */}
      <nav className="mobile-nav">
        <button
          className={`mobile-nav-item ${activePanel === 'agents' ? 'active' : ''}`}
          onClick={() => setActivePanel(activePanel === 'agents' ? 'none' : 'agents')}
        >
          <span className="mobile-nav-icon">&#x2B22;</span>
          <span>AGENTS</span>
        </button>
        <button
          className={`mobile-nav-item ${activePanel === 'earnings' ? 'active' : ''}`}
          onClick={() => setActivePanel(activePanel === 'earnings' ? 'none' : 'earnings')}
        >
          <span className="mobile-nav-icon">&#x2B50;</span>
          <span>EARNINGS</span>
        </button>
        <button
          className={`mobile-nav-item ${activePanel === 'settings' ? 'active' : ''}`}
          onClick={() => setActivePanel(activePanel === 'settings' ? 'none' : 'settings')}
        >
          <span className="mobile-nav-icon">&#x2699;</span>
          <span>SETTINGS</span>
        </button>
      </nav>

      {/* Bottom sheets for panels */}
      <BottomSheet
        isOpen={activePanel === 'agents'}
        onClose={() => setActivePanel('none')}
        title="MY AGENTS"
      >
        <AgentDashboard
          isOpen={true}
          onClose={() => setActivePanel('none')}
          onLocateAgent={(q, r) => {
            onLocateAgent(q, r);
            setActivePanel('none');
          }}
        />
      </BottomSheet>

      <BottomSheet
        isOpen={activePanel === 'earnings'}
        onClose={() => setActivePanel('none')}
        title="EARNINGS"
      >
        <Leaderboard />
      </BottomSheet>

      <BottomSheet
        isOpen={activePanel === 'settings'}
        onClose={() => setActivePanel('none')}
        title="SETTINGS"
      >
        <QualitySettings />
      </BottomSheet>
    </div>
  );
}

function QualitySettings() {
  const [quality, setQuality] = useState<'low' | 'medium' | 'high'>('medium');

  // Quality settings stored in localStorage and read by PerformanceAdapter
  const handleQualityChange = (level: 'low' | 'medium' | 'high') => {
    setQuality(level);
    localStorage.setItem('qualityLevel', level);
    // Dispatch event for PerformanceAdapter to pick up
    window.dispatchEvent(new CustomEvent('qualityChange', { detail: level }));
  };

  return (
    <div className="quality-panel">
      <div style={{ marginBottom: '12px', color: '#8B8B8B' }}>
        Graphics Quality
      </div>
      {(['low', 'medium', 'high'] as const).map((level) => (
        <button
          key={level}
          className={`quality-option ${quality === level ? 'selected' : ''}`}
          onClick={() => handleQualityChange(level)}
        >
          <span>{level.toUpperCase()}</span>
          <span style={{ color: '#5D8C3E' }}>
            {quality === level ? '✓' : ''}
          </span>
        </button>
      ))}
      <div style={{ marginTop: '12px', color: '#5A5A5A', fontSize: '6px' }}>
        Lower quality = better performance on older devices
      </div>
    </div>
  );
}
