**Create earningsStore.ts** with Zustand:

```typescript
import { create } from 'zustand';

interface EarningsState {
  claimable: bigint;
  totalClaimed: bigint;
  weightedScore: bigint;
  sharePercent: number;
  projectedDaily: bigint;
  rank: number | null;
  percentile: number;
  isLoading: boolean;
  lastUpdated: Date | null;

  // Actions
  setEarnings: (earnings: Partial<EarningsState>) => void;
  setLoading: (loading: boolean) => void;
  reset: () => void;
}

const initialState = {
  claimable: 0n,
  totalClaimed: 0n,
  weightedScore: 0n,
  sharePercent: 0,
  projectedDaily: 0n,
  rank: null,
  percentile: 0,
  isLoading: false,
  lastUpdated: null,
};

export const useEarningsStore = create<EarningsState>((set) => ({
  ...initialState,
  setEarnings: (earnings) => set({ ...earnings, lastUpdated: new Date() }),
  setLoading: (isLoading) => set({ isLoading }),
  reset: () => set(initialState),
}));
```

**Create useEarnings.ts** hook:

```typescript
import { useEffect, useCallback } from 'react';
import { useEarningsStore } from '../stores/earningsStore';
import { useWalletStore } from '../stores/walletStore';
import { socket } from '../lib/socket'; // or wherever socket is exported

export function useEarnings() {
  const { isAuthenticated } = useWalletStore();
  const { setEarnings, setLoading, ...state } = useEarningsStore();

  const fetchEarnings = useCallback(async () => {
    if (!isAuthenticated) return;

    setLoading(true);
    try {
      const res = await fetch('/api/earnings', { credentials: 'include' });
      if (!res.ok) throw new Error('Failed to fetch earnings');

      const data = await res.json();
      setEarnings({
        claimable: BigInt(data.claimable),
        totalClaimed: BigInt(data.totalClaimed),
        weightedScore: BigInt(data.weightedScore),
        sharePercent: data.sharePercent,
        projectedDaily: BigInt(data.projectedDaily),
        rank: data.rank,
        percentile: data.percentile,
      });
    } catch (err) {
      console.error('Error fetching earnings:', err);
    } finally {
      setLoading(false);
    }
  }, [isAuthenticated, setEarnings, setLoading]);

  // Subscribe to socket updates
  useEffect(() => {
    if (!isAuthenticated) return;

    socket.on('earnings:update', (data) => {
      setEarnings({
        claimable: BigInt(data.claimable),
        weightedScore: BigInt(data.weightedScore),
        projectedDaily: BigInt(data.projected24h),
        rank: data.rank,
        percentile: data.percentile,
      });
    });

    return () => {
      socket.off('earnings:update');
    };
  }, [isAuthenticated, setEarnings]);

  // Initial fetch
  useEffect(() => {
    fetchEarnings();
  }, [fetchEarnings]);

  return { ...state, refetch: fetchEarnings };
}
```

**Create useClaimEarnings.ts** hook:

```typescript
import { useState, useCallback } from 'react';
import { useConnection, useWallet } from '@solana/wallet-adapter-react';
import { Transaction } from '@solana/web3.js';

export function useClaimEarnings() {
  const { connection } = useConnection();
  const wallet = useWallet();
  const [isClaiming, setIsClaiming] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const claim = useCallback(async () => {
    if (!wallet.publicKey || !wallet.signTransaction) {
      setError('Wallet not connected');
      return null;
    }

    setIsClaiming(true);
    setError(null);

    try {
      // 1. Get claim transaction from server
      const res = await fetch('/api/earnings/claim', {
        method: 'POST',
        credentials: 'include',
      });

      if (!res.ok) {
        const { error } = await res.json();
        throw new Error(error);
      }

      const { transaction, amount } = await res.json();

      // 2. Deserialize and sign
      const tx = Transaction.from(Buffer.from(transaction, 'base64'));
      const signed = await wallet.signTransaction(tx);

      // 3. Send and confirm
      const signature = await connection.sendRawTransaction(signed.serialize());
      await connection.confirmTransaction(signature, 'confirmed');

      // 4. Confirm with server
      await fetch('/api/earnings/confirm', {
        method: 'POST',
        credentials: 'include',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ signature, amount }),
      });

      return { signature, amount };
    } catch (err) {
      const message = err instanceof Error ? err.message : 'Claim failed';
      setError(message);
      return null;
    } finally {
      setIsClaiming(false);
    }
  }, [wallet, connection]);

  return { claim, isClaiming, error };
}
```
