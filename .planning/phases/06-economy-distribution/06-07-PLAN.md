```typescript
import { useState, useEffect, useCallback } from 'react';

interface LeaderboardEntry {
  wallet: string;
  displayName: string;
  score: bigint;
  rank: number;
}

interface LeaderboardData {
  top10: LeaderboardEntry[];
  userRank: number | null;
  userScore: bigint;
  totalUsers: number;
}

export function useLeaderboard() {
  const [data, setData] = useState<LeaderboardData | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  const fetchLeaderboard = useCallback(async () => {
    setIsLoading(true);
    try {
      const res = await fetch('/api/leaderboard', { credentials: 'include' });
      if (!res.ok) throw new Error('Failed to fetch leaderboard');

      const json = await res.json();
      setData({
        top10: json.top10.map((e: any) => ({
          ...e,
          score: BigInt(e.score),
        })),
        userRank: json.userRank ?? null,
        userScore: json.userScore ? BigInt(json.userScore) : 0n,
        totalUsers: json.totalUsers,
      });
    } catch (err) {
      console.error('Leaderboard fetch error:', err);
    } finally {
      setIsLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchLeaderboard();
    // Refresh every 30 seconds
    const interval = setInterval(fetchLeaderboard, 30_000);
    return () => clearInterval(interval);
  }, [fetchLeaderboard]);

  // TODO: Add socket subscription for leaderboard:update

  return { data, isLoading, refetch: fetchLeaderboard };
}
```

**Heat-to-color function:**
```typescript
function heatToColor(heat: number): THREE.Color {
  // Blue (cold) -> Cyan -> Green -> Yellow -> Red (hot)
  const color = new THREE.Color();

  if (heat < 0.25) {
    color.setHSL(0.6 - heat * 0.4, 0.8, 0.5);  // Blue to Cyan
  } else if (heat < 0.5) {
    color.setHSL(0.4 - (heat - 0.25) * 0.4, 0.8, 0.5);  // Cyan to Green
  } else if (heat < 0.75) {
    color.setHSL(0.2 - (heat - 0.5) * 0.4, 0.8, 0.5);  // Green to Yellow
  } else {
    color.setHSL(0.1 - (heat - 0.75) * 0.4, 0.9, 0.5);  // Yellow to Red
  }

  return color;
}
```

**Implementation:**
- Use flat hex-shaped geometry (or simple plane) for overlay
- Position overlay slightly above hexes (y + 0.01)
- Set instance transforms from hex q,r coordinates using hexToWorld
- Build instanceColor attribute from heat values
- Animate opacity for fade in/out using useFrame

```typescript
import { useRef, useMemo, useEffect } from 'react';
import * as THREE from 'three';
import { useFrame } from '@react-three/fiber';
import { hexToWorld, createFlatHexGeometry } from '../hex/hexMesh';

export function HeatMapOverlay({ hexData, visible }: HeatMapOverlayProps) {
  const meshRef = useRef<THREE.InstancedMesh>(null);
  const materialRef = useRef<THREE.MeshBasicMaterial>(null);

  // Build colors and matrices
  const { colors, matrices } = useMemo(() => {
    const colorArray = new Float32Array(hexData.length * 3);
    const matrixArray: THREE.Matrix4[] = [];
    const dummy = new THREE.Object3D();

    hexData.forEach((hex, i) => {
      // Color
      const color = heatToColor(hex.heat);
      colorArray[i * 3] = color.r;
      colorArray[i * 3 + 1] = color.g;
      colorArray[i * 3 + 2] = color.b;

      // Position
      const [x, z] = hexToWorld(hex.q, hex.r);
      dummy.position.set(x, 0.1, z);  // Slightly above terrain
      dummy.updateMatrix();
      matrixArray.push(dummy.matrix.clone());
    });

    return { colors: colorArray, matrices: matrixArray };
  }, [hexData]);

  // Update instance matrices and colors
  useEffect(() => {
    const mesh = meshRef.current;
    if (!mesh) return;

    matrices.forEach((m, i) => mesh.setMatrixAt(i, m));
    mesh.instanceMatrix.needsUpdate = true;

    mesh.instanceColor = new THREE.InstancedBufferAttribute(colors, 3);
  }, [matrices, colors]);

  // Fade animation
  useFrame(() => {
    if (!materialRef.current) return;
    const target = visible ? 0.5 : 0;
    materialRef.current.opacity += (target - materialRef.current.opacity) * 0.1;
  });

  if (hexData.length === 0) return null;

  return (
    <instancedMesh ref={meshRef} args={[undefined, undefined, hexData.length]}>
      <circleGeometry args={[0.45, 6]} />  {/* Hex-shaped circle */}
      <meshBasicMaterial
        ref={materialRef}
        transparent
        opacity={0}
        depthWrite={false}
        vertexColors
      />
    </instancedMesh>
  );
}
```
