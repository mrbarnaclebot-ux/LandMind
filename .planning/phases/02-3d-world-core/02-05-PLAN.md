2. Implement createHexMaterial using ShaderMaterial for instance colors:
   ```typescript
   import { ShaderMaterial, Scene, Effect } from '@babylonjs/core';

   // Register shaders
   Effect.ShadersStore["hexVertexShader"] = `
     precision highp float;

     // Attributes
     attribute vec3 position;
     attribute vec3 normal;

     // Instance attributes
     attribute vec4 color;

     // Uniforms
     uniform mat4 viewProjection;
     uniform mat4 world;

     // Varyings
     varying vec3 vNormal;
     varying vec4 vColor;

     #include<instancesDeclaration>

     void main() {
       #include<instancesVertex>
       vec4 worldPos = finalWorld * vec4(position, 1.0);
       gl_Position = viewProjection * worldPos;
       vNormal = normalize((finalWorld * vec4(normal, 0.0)).xyz);
       vColor = color;
     }
   `;

   Effect.ShadersStore["hexFragmentShader"] = `
     precision highp float;

     varying vec3 vNormal;
     varying vec4 vColor;

     uniform vec3 lightDirection;
     uniform vec3 ambientColor;

     void main() {
       // Simple cell shading with 3 bands
       float ndl = dot(normalize(vNormal), normalize(lightDirection));
       float lightIntensity;

       if (ndl > 0.6) {
         lightIntensity = 1.0;
       } else if (ndl > 0.2) {
         lightIntensity = 0.7;
       } else {
         lightIntensity = 0.4;
       }

       vec3 finalColor = vColor.rgb * lightIntensity + ambientColor * 0.2;
       gl_FragColor = vec4(finalColor, 1.0);
     }
   `;
   ```

3. createHexMaterial function:
   ```typescript
   export function createHexMaterial(scene: Scene, name: string): ShaderMaterial {
     const mat = new ShaderMaterial(name, scene, {
       vertex: "hex",
       fragment: "hex"
     }, {
       attributes: ["position", "normal", "color"],
       uniforms: ["world", "viewProjection", "lightDirection", "ambientColor"]
     });

     // Default light direction (sun from upper right)
     mat.setVector3("lightDirection", new Vector3(1, 2, 1));
     mat.setColor3("ambientColor", new Color3(0.2, 0.2, 0.3));

     return mat;
   }
   ```

4. Export createHexMaterial

NOTE: If custom shader is too complex, fallback to StandardMaterial with a single color and accept that all hexes will be same color initially. Per-instance colors can be added as enhancement.
